13. Determine the author whose books have the highest combined stock.

USE assignment;
SELECT Name
FROM Authors
JOIN Books ON Books.AuthorID = Authors.AuthorID
WHERE Authors.AuthorID = (
    SELECT AuthorID as author
    FROM Books
    GROUP BY AuthorID
    ORDER BY SUM(Stock) DESC
    LIMIT 1
);
